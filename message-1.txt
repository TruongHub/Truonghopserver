import json
import requests
import os
import re
import time
import asyncio
import uvicorn
from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from threading import Thread
import logging
import sys

# Ép Terminal hỗ trợ tiếng Việt (Sửa lỗi UnicodeEncodeError)
if sys.platform == "win32":
    sys.stdout.reconfigure(encoding='utf-8')

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s - %(levelname)s - %(message)s',
    handlers=[
        logging.FileHandler('log.txt', encoding='utf-8'),
        logging.StreamHandler(sys.stdout)
    ]
)
logger = logging.getLogger(__name__)
class JobId(BaseModel):
    job_id: str
app = FastAPI()
PORT = 21411
API_BASE_URL = f"http://localhost:{PORT}"
# Token của bạn
DISCORD_TOKEN = "Dua token acc co maru premium cua ong vao day" # video huong dan : https://www.youtube.com/watch?v=sPKJOYXQdPw

MAX_JOBS = 10
job_ids = {
    "mirage": [], "nuilua": [], "haki": [], "sword": [],
    "fullmoon": [], "rareboss": [], "nearmoon": [], "commonboss": []
}

@app.post("/api/hop/{endpoint}")
async def post_job_id(endpoint: str, job: JobId):
    if endpoint not in job_ids:
        raise HTTPException(status_code=404, detail="Endpoint khong ton tai")
    
    # Kiểm tra nếu job_id đã tồn tại thì không thêm trùng
    if not any(item["job_id"] == job.job_id for item in job_ids[endpoint]):
        # Chèn Job ID mới vào ĐẦU danh sách (vị trí 0)
        job_ids[endpoint].insert(0, {"job_id": job.job_id, "time": time.strftime("%H:%M:%S")})
        
        # Nếu vượt quá 10 cái, xóa cái cũ nhất ở cuối
        if len(job_ids[endpoint]) > MAX_JOBS:
            job_ids[endpoint].pop()
            
    return {"status": "success"}

@app.get("/api/hop/{endpoint}")
async def get_job_id(endpoint: str):
    if endpoint not in job_ids:
        return {"error": "Endpoint khong ton tai"}
    
    # Trả về toàn bộ danh sách 10 cái để hiển thị trên web
    return {
        "category": endpoint,
        "total": len(job_ids[endpoint]),
        "jobs": job_ids[endpoint]
    }

def get_latest_messages(channel_id, token):
    headers = {'Authorization': token}
    url = f'https://discord.com/api/v9/channels/{channel_id}/messages?limit=5'
    try:
        res = requests.get(url, headers=headers, timeout=10)
        if res.status_code == 200: return res.json()
    except: pass
    return []

def extract_job_id(message_data):
    """Quét tất cả nội dung để tìm Job ID"""
    # Tìm trong Embeds
    for embed in message_data.get('embeds', []):
        # Quét trong Fields
        for field in embed.get('fields', []):
            val = field.get('value', '')
            # Regex tìm UUID (định dạng của Job ID)
            match = re.search(r'[a-f0-9]{8}-[a-f0-9]{4}-[a-f0-9]{4}-[a-f0-9]{4}-[a-f0-9]{12}', val)
            if match: return match.group(0)
        
        # Quét trong Description
        desc = embed.get('description', '')
        match = re.search(r'[a-f0-9]{8}-[a-f0-9]{4}-[a-f0-9]{4}-[a-f0-9]{4}-[a-f0-9]{12}', desc)
        if match: return match.group(0)
            
    return None
BASE_DIR = os.path.dirname(os.path.abspath(__file__))
async def run_client():
    # Load danh mục từ api.json để biết channel nào thuộc boss nào
    try:
        api_config_path = os.path.join(BASE_DIR, 'api.json')
        with open(api_config_path, 'r', encoding='utf-8') as f:
            api_map = json.load(f)
    except:
        logger.error("Khong tim thay file api.json!")
        return

    seen_job_ids = {} # {job_id: timestamp}

    logger.info("Bot dang chay... Khong dung config.json")
    while True:
        for endpoint, data in api_map.items():
            for channel_id in data.get('ids', []):
                messages = get_latest_messages(channel_id, DISCORD_TOKEN)
                
                for msg in messages:
                    job_id = extract_job_id(msg)
                    if job_id:
                        now = time.time()
                        # Tránh trùng lặp trong 5 phút
                        if job_id not in seen_job_ids or (now - seen_job_ids[job_id] > 300):
                            seen_job_ids[job_id] = now
                            # Gửi lên API nội bộ
                            try:
                                requests.post(f"{API_BASE_URL}/api/hop/{endpoint}", json={"job_id": job_id})
                                logger.info(f"成功 [ {endpoint.upper()} ] -> {job_id}")
                            except: pass
        await asyncio.sleep(10)

if __name__ == "__main__":
    Thread(target=lambda: uvicorn.run(app, host="0.0.0.0", port=PORT, log_level="error")).start()
    asyncio.run(run_client())
